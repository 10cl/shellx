## AI Assistant Features

### Command Suggestions

The AI assistant provides intelligent command suggestions:

```javascript
// AI suggestion system
class AISuggestionManager {
  constructor() {
    this.suggestions = [];
    this.context = {};
  }

  async generateSuggestions(input) {
    const response = await fetch('/api/ai/suggestions', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        input: input,
        context: this.context
      })
    });

    const suggestions = await response.json();
    this.displaySuggestions(suggestions);
  }

  displaySuggestions(suggestions) {
    const suggestionsList = document.getElementById('suggestions-list');
    suggestionsList.innerHTML = '';

    suggestions.forEach(suggestion => {
      const li = document.createElement('li');
      li.innerHTML = `
        <div class="suggestion-item">
          <span class="suggestion-command">${suggestion.command}</span>
          <span class="suggestion-description">${suggestion.description}</span>
          <button onclick="applySuggestion('${suggestion.command}')">Apply</button>
        </div>
      `;
      suggestionsList.appendChild(li);
    });
  }
}
```

## Command History Visualization

### Interactive Command History

Visual representation of command history with AI insights:

```javascript
// Command history with visual timeline
class CommandHistoryVisualizer {
  constructor() {
    this.history = [];
    this.container = document.getElementById('history-container');
  }

  addCommand(command, result) {
    const historyItem = {
      id: Date.now(),
      command: command,
      result: result,
      timestamp: new Date(),
      success: result.success
    };

    this.history.unshift(historyItem);
    this.renderHistory();
  }

  renderHistory() {
    this.container.innerHTML = '';
    
    this.history.forEach(item => {
      const historyElement = document.createElement('div');
      historyElement.className = `history-item ${item.success ? 'success' : 'error'}`;
      historyElement.innerHTML = `
        <div class="history-header">
          <span class="timestamp">${item.timestamp.toLocaleTimeString()}</span>
          <span class="status ${item.success ? 'success' : 'error'}">
            ${item.success ? '✓' : '✗'}
          </span>
        </div>
        <div class="command">${item.command}</div>
        <div class="result-preview">${this.truncateOutput(item.result.output)}</div>
        <div class="actions">
          <button onclick="repeatCommand('${item.command}')">Repeat</button>
          <button onclick="explainCommand('${item.command}')">Explain</button>
        </div>
      `;
      
      this.container.appendChild(historyElement);
    });
  }

  truncateOutput(output) {
    return output.length > 100 ? output.substring(0, 100) + '...' : output;
  }
}
```

### Performance Monitoring Script

```javascript
// Performance monitoring
class PerformanceMonitor {
  constructor() {
    this.metrics = {
      aiResponseTimes: [],
      commandExecutionTimes: [],
      successRate: 1.0
    };
    this.startTime = null;
  }

  startCommand() {
    this.startTime = performance.now();
  }

  endCommand(success) {
    const endTime = performance.now();
    const duration = endTime - this.startTime;
    
    this.metrics.commandExecutionTimes.push(duration);
    this.updateSuccessRate(success);
    this.updateUI();
  }

  recordAIResponse(responseTime) {
    this.metrics.aiResponseTimes.push(responseTime);
    this.updateUI();
  }

  updateSuccessRate(success) {
    const recentCommands = this.metrics.commandExecutionTimes.slice(-10);
    const successCount = recentCommands.filter(cmd => cmd.success).length;
    this.metrics.successRate = successCount / recentCommands.length;
  }

  updateUI() {
    // Update response time
    const avgResponseTime = this.getAverage(this.metrics.aiResponseTimes);
    document.getElementById('ai-response-time').textContent = `${avgResponseTime}ms`;
    
    // Update execution time
    const avgExecutionTime = this.getAverage(this.metrics.commandExecutionTimes);
    document.getElementById('command-execution').textContent = `${avgExecutionTime}ms`;
    
    // Update success rate
    const successPercentage = Math.round(this.metrics.successRate * 100);
    document.getElementById('success-rate').textContent = `${successPercentage}%`;
    
    // Update progress bar
    const progressBar = document.getElementById('success-progress');
    progressBar.style.width = `${successPercentage}%`;
  }

  getAverage(array) {
    if (array.length === 0) return 0;
    const sum = array.reduce((a, b) => a + b, 0);
    return Math.round(sum / array.length);
  }
}
```
